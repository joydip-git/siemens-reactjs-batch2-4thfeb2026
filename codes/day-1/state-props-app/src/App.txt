import { useState } from "react";

//import { Component, type ReactNode } from "react";

type StateType = {
  counter: number,
  name?: string
}
const App = () => {
  console.log('render');
  //let counter = 0
  const [state, setState] = useState<StateType>({ counter: 0, name: 'joydip' })
  const counterHandler = () => {
    console.log(state);
    //counter = counter + 1
    //setState({ ...state, counter: 100 })
    setState(
      (currentState) => {
        return {
          ...currentState,
          counter: currentState.counter + 1
        }
      }
    )
    console.log(state);
  }
  return (
    <div>
      <span>Counter: &nbsp; {state.counter}</span>
      <br />
      <button type="button" onClick={counterHandler}>Increase</button>
    </div>
  )
}
